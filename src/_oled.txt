void LcdStatus::showReadings(float distance, float level, float smoothLevel) {
  // 1. Hitung Persentase Ketinggian Air
  // Menggunakan maxDistance = 8 cm dari konfigurasi Anda
  float maxDist = 8.0; 
  float percentage = (smoothLevel / maxDist) * 100.0;

  // 2. Tampilkan nilai ke layar (Sesuaikan dengan library display Anda)
  // Ini contoh jika menggunakan library Adafruit_SSD1306
  display.clearDisplay();
  display.setCursor(0, 0);
  display.print("Level: ");
  display.print(smoothLevel, 1);
  display.print(" cm");

  // 3. Logika Peringatan (Berdasarkan Persentase)
  display.setCursor(0, 20);
  
  if (percentage < 30.0) {
    // Air di bawah 30% dari kapasitas maksimal
    display.print("Status: AMAN");
    
  } else if (percentage >= 30.0 && percentage < 70.0) {
    // Air antara 30% - 70%
    display.print("Status: WASPADA");
    
  } else {
    // Air di atas 70% dari kapasitas maksimal
    // Tambahkan efek kedip (blink) untuk OLED monokrom
    if (millis() % 1000 < 500) { 
      display.print("Status: BAHAYA!");
    }
  }

  display.display();
}
